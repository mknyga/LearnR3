#'

#' Import on participants cgm
#'
#' @param file_path Path to file
#'
#' @returns Returns a data frame
#'
import_cgm <- function(file_path) {
  cgm <- file_path %>%
    readr::read_csv(
      show_col_types = FALSE,
      name_repair = snakecase::to_snake_case,
      n_max = 100
    )
  return(cgm)
}

    read_csv(
